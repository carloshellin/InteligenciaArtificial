(define (t-agujeros t jugador)

(define (obtener-posiciones posiciones indice lista jugador-actual)
  (cond
    [(empty? lista) posiciones]
    [(= (car lista) 0) (obtener-posiciones posiciones (+ indice 1) (cdr lista) jugador-actual)]
